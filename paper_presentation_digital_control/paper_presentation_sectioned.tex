\begin{frame}{Introduction: Motivation and Challenges}
  \begin{columns}[T]
    \begin{column}{0.58\textwidth}
      \small
      \textbf{Context}
      \begin{itemize}
        \item Wireless standards such as \textbf{IEEE 802.15.4g} need \textbf{high-resolution} ADCs at MS/s rates with \textbf{very low power}.
        \item SAR ADCs are attractive because they already offer excellent \textbf{energy efficiency}.
      \end{itemize}

      \textbf{Main problems above 10 bits}
      \begin{itemize}
        \item \textbf{DAC mismatch}: intrinsic capacitor matching in modern CMOS is practically limited to about \textbf{10--12 bits}.
        \item \textbf{Noise and comparator offset}: these begin to limit usable resolution.
        \item Larger capacitors help matching, but increase \textbf{power} and \textbf{area}, and can reduce \textbf{speed}.
      \end{itemize}
    \end{column}

      \centering

\begin{frame}{Introduction: Prior Fixes and This Paper's Solution}
  \begin{columns}[T]
    \begin{column}{0.60\textwidth}
      \small
      \textbf{Why prior fixes are costly}
      \begin{itemize}
        \item \textbf{DSP / LMS} and \textbf{statistics-based calibration} add digital processing, sample collection, or foreground time $\rightarrow$ more \textbf{area}, \textbf{power}, and \textbf{latency}.
        \item \textbf{Two-stage pipelined SAR + amplifier} can relax comparator noise, but the amplifier is hard to design and introduces new analog errors.
        \item \textbf{Comparator oversampling} improves reliability, but extra comparisons reduce \textbf{speed}.
      \end{itemize}

      \textbf{This paper proposes}
      \begin{itemize}
        \item A \textbf{13-bit SAR ADC} with a \textbf{reconfigurable comparator}.
        \item \textbf{Low-power on-chip background calibration} for \textbf{DAC mismatch} and \textbf{dynamic comparator offset}.
      \end{itemize}
    \end{column}

\begin{frame}{Section II: Design Philosophy}
  \begin{columns}[T]
    \begin{column}{0.58\textwidth}
      \small
      \begin{itemize}
        \item \textbf{Background} calibration can keep running during normal ADC operation; foreground methods usually need special conditions.
        \item Prior work often detects errors \textbf{indirectly}: analyze final ADC output and adapt coefficients with DSP/LMS-like logic.
        \item This paper detects errors \textbf{directly}: determine only the \textbf{sign} of the error, then step correction in the opposite direction.
        \item Intuition: a \textbf{hardware-efficient sign-gradient update} with much less arithmetic.
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering
      \begin{tikzpicture}[node distance=6mm]
        \node[box, fill=softred, minimum width=3.8cm] (ind1) {Indirect method\\observe final ADC output};

\begin{frame}{Section II: ADC Architecture (Fig. 2)}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \small
      \begin{itemize}
        \item The S\&H samples the \textbf{differential input} on the capacitive DAC array.
        \item Comparator + SAR logic + \textbf{monotonic switching} perform the successive-approximation search.
        \item The ADC uses \textbf{15 cycles} for a \textbf{13-bit} conversion, with \textbf{2 redundant cycles}.
        \item The \textbf{7th} redundant cycle helps DAC settling / mismatch calibration; the \textbf{11th} helps tolerate comparator noise / offset.
        \item A digital adder maps the \textbf{15-bit raw code} to the final \textbf{13-bit output}; an optional \textbf{16th cycle} is used for calibration.
      \end{itemize}
    \end{column}

\begin{frame}{Section II: Sign-Based Calibration Loops}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \small
      \textbf{Comparator offset detection}
      \begin{itemize}
        \item Add the \textbf{16th cycle}; keep the DAC code the same as in cycle 15.
        \item Switch the comparator from fine mode back to coarse mode.
        \item Compare \(D_{15}\) and \(D_{16}\) to get the \textbf{sign} of the offset error.
      \end{itemize}

      \textbf{DAC mismatch detection}
      \begin{itemize}
        \item Redundancy means two different internal DAC codes can represent the same final output.
        \item If code \(A\) is observed, the controller enables the 16th cycle and forces the alternate code \(B\).
        \item Comparing cycle 15 under \(A\) and cycle 16 under \(B\) reveals the \textbf{sign} of the mismatch.
      \end{itemize}

    \end{column}

\end{frame}

\begin{frame}{Section III: Low-Power Calibration Logic}
  \begin{columns}[T]
    \begin{column}{0.58\textwidth}
      \small
      \textbf{How they made the controller cheap}
      \begin{itemize}
        \item \textbf{Dynamic Sense\&Force}: code-detection logic mostly stays reset/inactive unless a calibration trigger appears.
        \item \textbf{Low activation rate}: specific \(X\)- and \(Y\)-code patterns reduce calibration activity to about \textbf{1.4\%} of conversions.
        \item \textbf{Digital LPF}: six \textbf{6-bit bidirectional counters} are used, one comparator path and five DAC paths.
        \item Counters move \textbf{up/down} based on the error sign and only update the calibration register at \((0, 63)\).
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering
      \begin{tikzpicture}[node distance=6mm]

\begin{frame}{Section III: Analog Correction Hardware and Measured Impact}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \small
      \textbf{Analog correction hardware}
      \begin{itemize}
        \item \textbf{DAC calibration}: custom \textbf{75 aF} capacitors provide \textbf{0.25 LSB} correction steps for the first five DAC capacitors.
        \item \textbf{Comparator correction}: programmable capacitor banks \(C_a\) and \(C_b\) create a compensating input step.
        \item Comparator correction range is up to \(\pm 63\) LSB with \textbf{1 LSB} steps.
      \end{itemize}

      \textbf{Takeaway}
      \begin{itemize}
        \item Tiny analog correction elements plus sparse digital control let the calibration overhead stay very small.
      \end{itemize}
